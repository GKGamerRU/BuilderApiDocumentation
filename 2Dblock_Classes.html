class Block
 public Vector2 Position;
 public Vector2 Scale;
 public Image Texture;
 Graphics _GFX;
 public int _IndexTexture;
 public string Mod, code;
 public string BlockModName;
 public Block(ref Graphics _Parent, Image _texture, Point _pos, int index, string _mod, string blockName, string _code)
 {
  BlockModName = blockName;
  Mod = _mod;
  Position = new Vector2(_pos.X, _pos.Y);
  Texture = _texture;
  _IndexTexture = index;
  _GFX = _Parent; draw();
  code = _code;
 }
 public void Execute(string code)
 {
  if (code != null && code !="")
  {
  CharpExecuter cs;
  cs = new CharpExecuter(new ExecuteLogHandler(delegate { }));
  cs.FormatSources(code);
  cs.Execute();
  }
 }
 public void draw() { _GFX.DrawImage(Texture, Position.x, Position.y); }
